fun divide(x : int, y : int): int
   ac : int;
   aux : int
begin
   if (x < 0)
      return 0
   fi;
   if (y <=0)
      return 0
   else
   begin
      ac <- 0;
      aux <- y;
      while (aux >=x) 
      begin
         aux <- aux - x;
	      ac <- ac + 1
       end
    end;
    return ac
end
