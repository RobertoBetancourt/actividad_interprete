fun factorial(x: int): int
  acumulador : int;
  aux : int
begin
   acumulador <- 1;
   aux <- 1;
   while (aux <= x)
   begin
      acumulador <- acumulador * aux;
      aux <- aux + 1
   end;
   return acumulador
end
