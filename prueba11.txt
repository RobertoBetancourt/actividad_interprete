  x : int;
  y : int;
  res : int;
  aux: int
begin
   read x;
   read y;

   res <- 0;

   if (x < 0)
   begin
      print x;
      print y
   end
   else
   begin
      if (x < y)
      begin
         aux <- x;
	 x <- y;
	 y <- res;
	 aux <- 0
      end
      fi;
      aux <- x;
      while (aux > 0)
      begin
         res <- res + x;
	       aux <- aux - 1
      end;
      print res;
   end
end
